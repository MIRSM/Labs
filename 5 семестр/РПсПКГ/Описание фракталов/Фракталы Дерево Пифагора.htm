Фракталы: Дерево Пифагора

Program DPif;
Uses Graph, CRT;
Const
  max = 3;
Var
 gd, gm : Integer;
Procedure lineto1(x, y : Integer; l, u : real);
Begin
   Line(x, y, Round(x + l * cos(u)), Round(y - l * sin(u)));
End;

Procedure Draw(x, y : Integer; l, u : real);
Begin
   If KeyPressed then exit;
   If l > max then Begin
      l := l * 0.7;
      lineto1(x, y, l, u);
      x := Round(x + l * cos(u));
      y := Round(y - l * sin(u));
      Draw(x, y, l, u + pi / 4);{Угол поворота 1}
      Draw(x, y, l, u - pi / 6);{Угол поворота 2}
   End;
End;

Begin
   gd := Detect;
   InitGraph(gd, gm, 'c:\bp\bgi');
   Draw(320, 460, 200, pi/2);
   ReadKey;
   CloseGraph;
End.

Не забудте в процедуре инициализации графического режима указать путь к графическому драйверу!

Приведенные примеры получены с углами поворота 45-45, 30-45 градусов соответственно. При этом в первом случае мы получаем нормальное обнаженное дерево Пифагора, а во втором обдуваемое ветром дерево.
