Фракталы: Биоморфы

program Bio;
uses graph, crt;
type
   complex = record X:real;Y:real; end;
const
   iter=50;
   max=70;
var
   z,z1,c:complex;
   x,y,n:integer;
   cancel:boolean;
   gd,gm : Integer;
begin
   cancel:=false;
   Randomize;
   GD:= Detect;
   InitGraph(GD,GM,'c:\bp\bgi');
   for y:=-getmaxy div 2 to getmaxy div 2 do
      for x:=-getmaxx div 2 to getmaxx div 2 do begin
         n:=0;
         z.X:=X*0.01;
         z.Y:=Y*0.01;
         c.X:=1.00003;
         c.Y:=1.01828;
         while ((sqr(z.x) < max) and (sqr(z.y) < max)) and (n < iter) do begin
            z1:=z;
            Z.X:=Z1.X*Z1.X*Z1.X*Z1.X + Z1.Y*Z1.Y*Z1.Y*Z1.Y - 6*Z1.X*Z1.X*Z1.Y*Z1.Y;
            Z.Y:=4*Z1.X*Z1.X*Z1.X*Z1.Y - 4*Z1.X*Z1.Y*Z1.Y*Z1.Y;
            Z1.X:=Z.X+C.X;
            Z1.Y:=Z.Y+C.Y;
            z:=z1;
            inc(n);
            if keypressed then cancel:=true;
         end;
         if (abs(z.x) > 10) or (abs(z.y) > 1000) then 
            putpixel(getmaxx div 2+x,getmaxy div 2+y,n mod 16);
         if cancel then exit;
      end;
   readkey;
   closegraph;
end.
Другие формулы Можно использовать следующие формулы:
  • z'=z4+c (приведена в программе)
    Re: x4 + y4 - 6x2y2 + a
    Im: 4x3y - 4xy3 + b,
    где z=x + yi, c=a + bi.

  • z'=sin(z) + ez + c
  • z'=z3 + c
    Re: x3 - 3y2x + a
    Im: 3x2y - y3 + b

  • z'=z5+ c
  • z'=sin(z) + z2 + c
    Re: sin(x)cosh(y) + x2 - y2 + a
    Im: cos(x)sinh(y) + 2xy + b,
    где sinh(x)=(ex - e-x)/2, cosh(x)=(ex + e-x)/2

  • z'=zz + z5 + c
  • z'=zz + z6 + c
