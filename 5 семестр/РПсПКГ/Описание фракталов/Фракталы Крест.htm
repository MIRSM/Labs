Фракталы: Крест

Этот фрактал получается применением двух преобразований на основе метода Монте-Карло. Эти преобразования выглядят следующим образом:
.
Здесь L - поворот на 90 градусов относительно точки (0, 0), а R - "растяжение" относительно точки (1, 0) с переменным показателем, который зависит от растояния от центра (1, 0). Приведенный пример получен при a=2,8. 
program Rnd;
Uses CRT, Graph;
Var
   gd,gm : Integer;
   
B>Procedure draw;
const
  iter = 70000;
  a = 2.8;
Var
   t, x, y, p : Real;
   k : LongInt;
   mx, my, rad : Integer;
Begin
   mx := 320;
   my := 240;
   rad := 200;
   Randomize;
   x := 0.0;
   y := 0.0;
   For k := 1 To iter Do Begin
      p := Random;
      t := x;
     If p <= 1/2 Then Begin
         x := -y;
         y := t;
      End
      Else
         Begin
            x := 1+(a*(x-1))/(sqr(x-1)+sqr(y)+1);
            y := a*y/(sqr(t-1)+sqr(y)+1);
         End;
      PutPixel(mx + Round(rad * x), my - Round(rad * y), 2);
   End;
End;

Begin
   gd := Detect;
   InitGraph(gd,gm,'c:\bp\bgi');
   draw;
   ReadKey;
   CloseGraph;
End.
