Фракталы: Двумерное множество Кантора

Программа для построения множества Кантора на языке Pascal выглядит следующим образом: 
Program Cantor3;
Uses CRT, Graph;
Const
  min = 2;
Var
  gd, gm : Integer;

Procedure Draw(x, y : Real; Size : Real);
Var
  s : Real;
Begin
  If size > min then Begin
    s := size / 2;
    Draw(x - size, y + size, s);
    Draw(x - size, y - size, s);
    Draw(x + size, y + size, s);
    Draw(x + size, y - size, s);
  End;
  Rectangle(Round(x - size), Round(y - size), 
     Round(x + size), Round(y + size));
  Bar(Round(x - size + 1), Round(y - size + 1), 
     Round(x + size - 1), Round(y + size - 1));
End;
Begin
  gd := Detect;
  InitGraph(gd, gm, '');
  setfillstyle(solidfill, 0);
  Draw(320, 240, 120);
  ReadKey;
  CloseGraph;
End.

Не забудте в процедуре инициализации графического режима указать путь к графическому драйверу!

Если конец программы переписать следующим образом, то получится весьма интересный эффект. 

...
begin
   gd := Detect;
   InitGraph(gd,gm,'c:\bp\bgi');
   setfillstyle(solidfill,0);
   setcolor(15);
   draw(320,240,120);
   readln;
   setwritemode(xorput);
   setcolor(0);
   draw(320,240,120);
   readln;
   closegraph;
end.


Треугольное двумерное множество Кантора.

Также можно построить треугольное двумерное множество Кантора. Это делает следующая программа. 
uses crt,graph;
const
  min_size=0;
Var
   gd,gm : Integer;
procedure triangle(xc,yc:Integer; l:integer);
Var
  x1,y1,x2,y2,x3,y3:Integer;
Begin
  x1:=xc;
  y1:=yc-l;
  x2:=xc+Round(l*cos(pi/6));
  y2:=yc+Round(l*sin(pi/6));
  x3:=xc-Round(l*cos(pi/6));
  y3:=yc+Round(l*sin(pi/6));
  SetColor(1);
  Line(x1,y1,x2,y2);
  Line(x2,y2,x3,y3);
  Line(x3,y3,x1,y1);
  FloodFill(xc,yc,1);
  SetColor(15);
  Line(x1,y1,x2,y2);
  Line(x2,y2,x3,y3);
  Line(x3,y3,x1,y1);
End;

procedure draw(x,y:integer;size:word);
var s:word;
begin
  if size>min_size then begin
    s:=size div 2;
    draw(x,y-size,s);
    draw(x+Round(size*cos(pi/6)), y+Round(size*sin(pi/6)),s);
    draw(x-Round(size*cos(pi/6)), y+Round(size*sin(pi/6)),s);
  end;
  triangle(x,y,size);
end;

begin
  gd := Detect;
  InitGraph(gd,gm,'c:\bp\bgi');
  draw(320,280,120);
  readln;
  closegraph;
end.
